							*onclick="changeCss();"*

							*onclick="displayURL();"*

							*onclick="this.select();"* (--> selektiert Text)
					.displayOut --> Button um Fenster zu schließen

				#backwardsBehind --> Bereich für zurückblätter-Button, Button schwach im Hintergrund
					#backwards --> Button um zurückzublättern
								*onclick="navigatePage(--pagenumber);"*

				#forwardBehind --> Bereich für vorblätter-Button, Button schwach im Hintergrund
					#forward --> Button um vorzublättern
								*onclick="navigatePage(++pagenumber);"*

				#pageCounter --> Bereich mit max Seitenzahl

				#inputPage --> Bereich für aktuelle Seitennummer
					#iptPgNr --> Seitennummer-Formular
								*onkeydown="input(event);"*

				#openThumbs --> Button um Overview / Seitenübersicht zu öffnen
							*onclick="showOverview();"*
							
				#chapterOpener --> Button um Kapitelübersicht zu öffnen
							*onclick="openChapter();"*
